Form Validation — Name & Email

Topics: HTML forms · JavaScript validation · DOM · if/else
📘 Concept Form validation checks user input before sending data to the server. We read the input value with document.getElementById(), then check conditions with if/else and show or hide error messages.
📄 Source Code
<!-- HTML Structure -->
<input type="text"  id="name"  placeholder="Enter your name" />
<input type="email" id="email" placeholder="Enter your email" />
<button onclick="validate()">Submit</button>
<div id="output"></div>

<!-- JavaScript Logic -->
<script>
  function validate() {
    let name  = document.getElementById("name").value.trim();
    let email = document.getElementById("email").value.trim();
    let out   = document.getElementById("output");

    // Rule 1: name must not be empty
    if (name === "") {
      out.textContent = "❌ Name is required.";
      return;
    }

    // Rule 2: email must contain "@"
    if (!email.includes("@")) {
      out.textContent = "❌ Enter a valid email.";
      return;
    }

    out.textContent = `✅ Hello, ${name}! Email ${email} is valid.`;
  }
</script>

Events — Button Click Counter

Topics: onclick event · variables · DOM update
📘 Concept An event is something that happens in the browser — like a button click. We attach a function to onclick and update a counter variable each time the user clicks. innerHTML lets us change the text shown on the page.
📄 Source Code
<p>Clicks: <span id="count">0</span></p>
<button onclick="addClick()">Click Me!</button>
<button onclick="reset()">Reset</button>

<script>
  let counter = 0;          // stores the count

  function addClick() {
    counter++;             // counter = counter + 1
    document.getElementById("count").innerHTML = counter;
  }

  function reset() {
    counter = 0;
    document.getElementById("count").innerHTML = counter;
  }
</script>

Arrays & Loops — Student Marks List

Topics: Array · for loop · if/else · grade logic
📘 Concept An array stores multiple values in one variable: let marks = [85, 92, 60]. A for loop repeats code for each item. We combine both to loop through student marks and calculate a grade for each student.
📄 Source Code
<script>
  // Array of student objects
  const students = [
    { name: "Ravi",   marks: 92 },
    { name: "Priya",  marks: 74 },
    { name: "Anil",   marks: 55 },
    { name: "Sneha",  marks: 88 },
    { name: "Kiran",  marks: 43 },
  ];

  function getGrade(m) {
    if (m >= 90) return "A+";
    if (m >= 75) return "A";
    if (m >= 60) return "B";
    if (m >= 45) return "C";
    return "Fail";
  }

  function showMarks() {
    let html = "<table><tr><th>Name</th><th>Marks</th><th>Grade</th></tr>";

    // for loop iterates over each student
    for (let i = 0; i < students.length; i++) {
      let s = students[i];
      let g = getGrade(s.marks);
      html += `<tr><td>${s.name}</td>
               <td>${s.marks}</td>
               <td>${g}</td></tr>`;
    }
    html += "</table>";
    document.getElementById("marksOut").innerHTML = html;
  }
</script>

Array .map() — Price Calculator with GST

Topics: Array.map() · arrow functions · template literals
📘 Concept Array.map() creates a new array by transforming each item. It does NOT change the original array. Syntax: newArray = oldArray.map(item => item * 2). Here we calculate prices after adding 18% GST to each product.
📄 Source Code
<script>
  const products = [
    { name: "Laptop",  price: 50000 },
    { name: "Phone",   price: 20000 },
    { name: "Tablet",  price: 15000 },
    { name: "Watch",   price: 8000  },
  ];

  // .map() returns a NEW array with GST added
  const withGST = products.map(item => ({
    ...item,                                  // copy original fields
    gstPrice: Math.round(item.price * 1.18)  // add 18% GST
  }));

  function showPrices() {
    let html = `<table>
      <tr><th>Product</th><th>Base Price</th>
          <th>+ 18% GST</th></tr>`;

    withGST.forEach(p => {
      html += `<tr>
        <td>${p.name}</td>
        <td>₹${p.price.toLocaleString()}</td>
        <td>₹${p.gstPrice.toLocaleString()}</td>
      </tr>`;
    });
    html += "</table>";
    document.getElementById("priceOut").innerHTML = html;
  }
</script>

Registration Form — Show Data on Next "Page"

Topics: Form input · localStorage · section switching · validation
📘 Concept We use two <div> sections to simulate "pages". After the user fills the form and clicks Submit, we hide Section 1 and show Section 2 with all entered data. localStorage saves data in the browser so it persists.
📄 Source Code
<!-- Page 1: Form -->
<div id="regForm">
  <input id="rName"  placeholder="Full Name"  />
  <input id="rEmail" placeholder="Email"      />
  <input id="rPhone" placeholder="Phone"      />
  <input id="rCity"  placeholder="City"       />
  <button onclick="submitReg()">Register</button>
</div>

<!-- Page 2: Summary (hidden initially) -->
<div id="regSummary" style="display:none">
  <h3>Registration Successful!</h3>
  <div id="summaryData"></div>
</div>

<script>
  function submitReg() {
    let data = {
      name:  document.getElementById("rName").value.trim(),
      email: document.getElementById("rEmail").value.trim(),
      phone: document.getElementById("rPhone").value.trim(),
      city:  document.getElementById("rCity").value.trim(),
    };

    // Validate required fields
    if (!data.name || !data.email) {
      alert("Name and Email are required!");
      return;
    }

    // Save to localStorage
    localStorage.setItem("regData", JSON.stringify(data));

    // Show summary page
    document.getElementById("regForm").style.display    = "none";
    document.getElementById("regSummary").style.display = "block";
    document.getElementById("summaryData").innerHTML =
      `<p>👤 ${data.name}</p>
       <p>📧 ${data.email}</p>
       <p>📞 ${data.phone}</p>
       <p>🏙️ ${data.city}</p>`;
  }
</script>

Login Validation with Static Data

Topics: Static credentials · string comparison · session state
📘 Concept In real apps, login is checked against a server. For learning we use static (hardcoded) credentials stored in a JavaScript object. The user's input is compared with === (strict equality). After login, we show a welcome panel and hide the login form.
💡 Demo Credentials:   Email: admin@test.com   Password: pass123
📄 Source Code
<script>
  // Static user data (in real apps this is on the server)
  const USERS = [
    { email: "admin@test.com",   password: "pass123", name: "Admin"  },
    { email: "student@test.com", password: "abc456",  name: "Ravi"   },
  ];

  function doLogin() {
    let email = document.getElementById("lEmail").value.trim();
    let pass  = document.getElementById("lPass").value;

    // Find a matching user
    let user = USERS.find(
      u => u.email === email && u.password === pass
    );

    if (user) {
      // Login success
      document.getElementById("loginForm").style.display = "none";
      document.getElementById("loginMsg").innerHTML =
        `✅ Welcome, <strong>${user.name}</strong>! You are logged in.`;
    } else {
      document.getElementById("loginErr").textContent =
        "❌ Invalid email or password.";
    }
  }
</script>

For Loop — Multiplication Table

Topics: for loop · user input · string building · innerHTML
📘 Concept A for loop runs a block of code a specific number of times. Syntax: for (let i = 1; i <= 10; i++) — start at 1, run while i ≤ 10, increment by 1 each time. We use it to generate the multiplication table for any number the user enters.
📄 Source Code
<script>
  function showTable() {
    let n   = parseInt(document.getElementById("tableNum").value);
    let out = document.getElementById("tableOut");

    if (isNaN(n) || n < 1) {
      out.textContent = "Please enter a valid number.";
      return;
    }

    let result = "";

    // for loop: runs 10 times (i = 1 to 10)
    for (let i = 1; i <= 10; i++) {
      result += `${n} × ${i} = ${n * i}\n`;
    }

    out.textContent = result;
  }
</script>

Error Handling — try / catch / finally

Topics: try · catch · finally · throw · typeof
📘 Concept When code might fail (e.g., dividing by zero, invalid JSON), we wrap it in try { }. If an error occurs, JavaScript jumps to catch (e) { } where e.message tells us what went wrong. finally { } always runs whether there was an error or not.
📄 Source Code
<script>
  function divideNumbers() {
    let a = parseFloat(document.getElementById("numA").value);
    let b = parseFloat(document.getElementById("numB").value);
    let out = document.getElementById("divOut");

    try {
      // Manually throw an error for division by zero
      if (isNaN(a) || isNaN(b)) {
        throw new Error("Both inputs must be valid numbers.");
      }
      if (b === 0) {
        throw new Error("Cannot divide by zero!");
      }

      let result = a / b;
      out.textContent = `✅ ${a} ÷ ${b} = ${result}`;
      out.className = "output-box";

    } catch (error) {
      // This runs when an error is thrown
      out.textContent = `❌ Error: ${error.message}`;
      out.className = "output-box error";

    } finally {
      // This ALWAYS runs
      console.log("Division attempted.");
    }
  }

  // JSON Parse error example
  function parseJSON() {
    let input = document.getElementById("jsonInput").value;
    let out   = document.getElementById("jsonOut");
    try {
      let data = JSON.parse(input);
      out.textContent = "✅ Valid JSON: " + JSON.stringify(data, null, 2);
      out.className = "output-box";
    } catch (e) {
      out.textContent = "❌ Invalid JSON: " + e.message;
      out.className = "output-box error";
    }
  }
</script>

Dynamic List — Add & Remove Items

Topics: Array push/splice · DOM createElement · event listeners · filter
📘 Concept We maintain a JavaScript array as the single source of truth for our list, then re-render the DOM every time the array changes. array.push(item) adds to the end. array.splice(index, 1) removes one item at a given position.
📄 Source Code
<script>
  let items = ["Buy milk", "Read a book", "Walk 30 min"];

  function renderList() {
    let ul = document.getElementById("myList");
    ul.innerHTML = "";          // clear old list

    if (items.length === 0) {
      ul.innerHTML = "<li style='color:#718096'>No items yet.</li>";
      return;
    }

    // forEach creates one <li> per item
    items.forEach((item, index) => {
      let li = document.createElement("li");
      li.innerHTML = `${item}
        <button onclick="removeItem(${index})">✕</button>`;
      ul.appendChild(li);
    });
  }

  function addItem() {
    let val = document.getElementById("newItem").value.trim();
    if (!val) { alert("Enter something first!"); return; }
    items.push(val);                      // add to array
    document.getElementById("newItem").value = "";
    renderList();                            // refresh DOM
  }

  function removeItem(index) {
    items.splice(index, 1);               // remove 1 item at index
    renderList();
  }

  renderList();   // run on page load
</script>

    Full Mini App — Register → Login → Dashboard

    Topics: Multi-step flow · localStorage · form validation · session management · all concepts combined
    📘 Concept This is the capstone program combining everything: registration form with full validation, saving user data to localStorage, login with credential check, a welcome dashboard showing registered info, and a logout button to reset the session. Three "pages" are managed by toggling display:none / block.
    📄 Source Code (key functions)
    <script>
  // ── REGISTER ──────────────────────────────
  function register() {
    let name  = getVal("fName");
    let email = getVal("fEmail");
    let pass  = getVal("fPass");
    let conf  = getVal("fConf");

    // Validation chain
    if (!name)               return showErr("Name is required");
    if (!email.includes("@")) return showErr("Invalid email");
    if (pass.length < 6)      return showErr("Password min 6 chars");
    if (pass !== conf)         return showErr("Passwords do not match");

    // Save user to localStorage
    localStorage.setItem("appUser",
      JSON.stringify({ name, email, pass }));

    navigate("loginPage");   // go to login
  }

  // ── LOGIN ─────────────────────────────────
  function login() {
    let saved = JSON.parse(localStorage.getItem("appUser"));
    if (!saved) return showLoginErr("Please register first");

    let email = getVal("lEmail");
    let pass  = getVal("lPass");

    if (email === saved.email && pass === saved.pass) {
      showDashboard(saved);
    } else {
      showLoginErr("Incorrect email or password");
    }
  }

  // ── NAVIGATE between pages ─────────────────
  function navigate(pageId) {
    ["regPage", "loginPage", "dashboard"].forEach(id => {
      document.getElementById(id).style.display =
        id === pageId ? "block" : "none";
    });
  }
</script>
