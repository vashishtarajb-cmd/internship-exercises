HTML — Structure & Markup

HTML (HyperText Markup Language) is the skeleton of every web page. It defines what content exists — JavaScript later decides what it does, and CSS decides how it looks.

1.1  The Boilerplate — Every file starts here HTML

Why
Browsers need a declaration of document type and character encoding before they can render anything correctly.
index.html
<!DOCTYPE html>                 <!-- Tells browser: "this is HTML5" -->
<html lang="en">
<head>
  <meta charset="UTF-8">         <!-- Support all characters -->
  <meta name="viewport" content="width=device-width, initial-scale=1.0">
  <title>My First Page</title>
  <link rel="stylesheet" href="style.css">    <!-- External CSS -->
</head>
<body>
  <!-- All visible content goes here -->
  <h1>Hello, World!</h1>

  <script src="app.js"></script>     <!-- JS at bottom so HTML loads first -->
</body>
</html>
💡 Tip — Why put <script> at the bottom?
Browsers read HTML top-to-bottom. If your script is at the top, it may try to access elements that haven't been created yet — and fail silently.

1.4  Div, ID & Class — the targeting system HTML

Why
id is a unique name — like a passport number (one element per page). class is a category — like a job title (many elements can share it). JavaScript and CSS use these to find and manipulate elements.
layout.html

  <h1>My App</h1>

<!-- CLASS: reusable grouping, used by JS with getElementsByClassName -->
<div class="card">Card One</div>
<div class="card">Card Two</div>
<div class="card highlight">Card Three (has two classes)</div>

<!-- DATA ATTRIBUTES: store custom values on elements -->

     data-category="electronics">
  Laptop

JavaScript — Behaviour & Logic

JavaScript makes pages interactive. It runs in the browser, reads and changes HTML, responds to user actions, and communicates with servers.

2.1  Variables, Types & Operators JS

How
Use const when the value won't change, let when it will. Avoid var in modern code — it has confusing scoping rules.
variables.js
// ── Declaring variables ──────────────────────────
const siteName  = "MyApp";          // string
let   pageCount = 0;              // number
let   isLoggedIn= false;          // boolean
let   userAge   = null;           // null (intentionally empty)
let   score;                      // undefined (declared, no value yet)

// ── Arithmetic operators ─────────────────────────
let x = 10, y = 3;
console.log(x + y);   // 13
console.log(x - y);   // 7
console.log(x * y);   // 30
console.log(x / y);   // 3.333…
console.log(x % y);   // 1   (remainder)

// ── Comparison operators ─────────────────────────
console.log(5 === 5);  // true  (strict equal)
console.log(5 ==  "5"); // true  (loose — avoids this)
console.log(5 !== 4);  // true  (strict not equal)

// ── String concatenation / template literal ──────
const msg1 = "Hello, " + siteName;
const msg2 = `Hello, ${siteName}! Pages visited: ${pageCount}`;
console.log(msg2);    // Hello, MyApp! Pages visited: 0

2.2  DOM — Finding & Changing Elements JS

Why
The DOM (Document Object Model) is the browser's live tree of your HTML. JavaScript reaches into this tree to read, change, add, or remove elements — without reloading the page.
dom.js
// ── Selecting elements ────────────────────────────
const el1  = document.getElementById('header');
const el2  = document.querySelector('#header');       // same as above
const cards = document.querySelectorAll('.card');     // all .card divs
const btns  = document.getElementsByTagName('button');

// ── Reading & writing content ────────────────────
const box = document.getElementById('myBox');
box.textContent  = "Plain text — safe from XSS";
box.innerHTML    = "<strong>Bold text</strong>";  // renders HTML

// ── Reading & changing attributes ───────────────
const link = document.querySelector('a');
console.log(link.getAttribute('href'));
link.setAttribute('href', 'https://example.com');

// ── Changing CSS styles ──────────────────────────
box.style.color      = 'red';
box.style.fontSize   = '18px';
box.style.display    = 'none';   // hide the element
box.style.display    = 'block';  // show it again

// ── Adding / removing CSS classes ───────────────
box.classList.add('active');
box.classList.remove('hidden');
box.classList.toggle('highlight');  // adds if absent, removes if present
box.classList.contains('active'); // → true / false

// ── Creating & inserting new elements ───────────
const newPara = document.createElement('p');
newPara.textContent = "I was created by JavaScript!";
document.body.appendChild(newPara);  // appended at the end

// ── Removing an element ──────────────────────────
const toRemove = document.getElementById('old');
toRemove.remove();

2.3  Scripting Div Sections JS

A very common pattern: show/hide panels, update a result box, or build a product card dynamically.

div-scripting.html + js
<!-- HTML -->
<button onclick="togglePanel()">Toggle Panel</button>

<div id="panel" style="display:none">
  I am a hidden panel!
</div>

<div id="result"></div>

<!-- JavaScript -->
<script>
function togglePanel() {
  const panel = document.getElementById('panel');
  if (panel.style.display === 'none') {
    panel.style.display = 'block';
  } else {
    panel.style.display = 'none';
  }
}

// Dynamically build a product card inside a div
function renderProduct(name, price) {
  const result = document.getElementById('result');
  result.innerHTML = `
    <div class="card">
      <h3>${name}</h3>
      <p>Price: ₹${price}</p>
      <button onclick="addToCart('${name}')">Add to Cart</button>
    </div>
  `;
}
renderProduct("Laptop", 49999);
</script>
🖥 Output
Clicking "Toggle Panel" shows/hides the div. The result div renders a product card with a live "Add to Cart" button.

2.4  Event Handlers — Reacting to user actions JS

Where to use
Button clicks, form submissions, keyboard input, mouse hover, page load — any user interaction.
events.js
// ── Method 1: Inline (simple, quick) ─────────────
// <button onclick="greet()">Greet</button>
function greet() {
  alert("Hello!");
}

// ── Method 2: DOM property (overwrites previous) ──
const btn = document.getElementById('myBtn');
btn.onclick = function() {
  console.log("Button clicked!");
};

// ── Method 3: addEventListener (best practice) ───
btn.addEventListener('click', function() {
  console.log("Click via addEventListener");
});

// ── Common events ────────────────────────────────
const input = document.getElementById('myInput');

input.addEventListener('input', (e) => {
  console.log("Typed:", e.target.value);  // live as you type
});

input.addEventListener('change', (e) => {
  console.log("Changed to:", e.target.value);  // fires on blur
});

input.addEventListener('keydown', (e) => {
  if (e.key === 'Enter') console.log("Enter pressed");
});

// ── Mouse events ─────────────────────────────────
const box = document.getElementById('hoverBox');
box.addEventListener('mouseover', () => box.style.background = 'yellow');
box.addEventListener('mouseout',  () => box.style.background = '');

// ── Page load event ──────────────────────────────
document.addEventListener('DOMContentLoaded', () => {
  console.log("Page is fully loaded and ready!");
});
💡 Event Object (e)
Every event listener receives an event object (often called e or event). It contains: e.target (the element that was clicked/typed), e.target.value (the input's current value), e.key (which key was pressed), and e.preventDefault() (stop the default action, e.g. form submission).

2.5  Form Validation — Custom JavaScript validation JS

validation.html + js
<!-- HTML -->
<form id="loginForm">
  <input type="text"     id="username" placeholder="Username">
  <input type="password" id="password" placeholder="Password">
  <div id="errorMsg" style="color:red"></div>
  <button type="submit">Login</button>
</form>

<!-- JavaScript -->
<script>
document.getElementById('loginForm').addEventListener('submit', function(e) {
  e.preventDefault();   // STOP the page from reloading

  const user  = document.getElementById('username').value.trim();
  const pass  = document.getElementById('password').value;
  const errEl = document.getElementById('errorMsg');

  // Rule 1: Required fields
  if (!user) {
    errEl.textContent = "Username is required.";
    return;
  }

  // Rule 2: Minimum length
  if (pass.length < 8) {
    errEl.textContent = "Password must be at least 8 characters.";
    return;
  }

  // Rule 3: Email format (using RegEx)
  const emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/;
  if (!emailPattern.test(user)) {
    errEl.textContent = "Enter a valid email address.";
    return;
  }

  // All checks passed
  errEl.textContent = "";
  console.log("Form is valid. Submitting…");
});
</script>
Output — Validation errors shown inline
⚠ Password must be at least 8 characters.

2.8  Arrays & Loops JS

arrays.js
const fruits = ['Apple', 'Banana', 'Cherry'];

// Read / modify
console.log(fruits[0]);            // Apple
fruits.push('Date');              // add to end
fruits.pop();                   // remove from end
fruits.unshift('Avocado');       // add to front
console.log(fruits.length);       // 4

// for loop
for (let i = 0; i < fruits.length; i++) {
  console.log(fruits[i]);
}

// for…of (cleaner)
for (const fruit of fruits) {
  console.log(fruit);
}

// forEach (callback)
fruits.forEach((f, i) => console.log(i, f));

// map — transform each element, returns new array
const upper = fruits.map(f => f.toUpperCase());
// → ['AVOCADO', 'APPLE', 'BANANA', 'CHERRY']

// filter — keep elements matching a condition
const longNames = fruits.filter(f => f.length > 5);
// → ['Avocado', 'Banana', 'Cherry']

// find — first element matching condition
const found = fruits.find(f => f.startsWith('B'));  // "Banana"

2.9  Objects & JSON JS

objects.js
// ── Creating an object ───────────────────────────
const user = {
  id:    1,
  name:  "Hari",
  email: "hari@example.com",
  role:  "developer",
  greet: function() {             // method

delete user.email;               // remove key

// ── JSON ──────────────────────────────────────────
// JSON is a string format used to send data to/from servers

// Object → JSON string (serialise)
const jsonString = JSON.stringify(user);
// → '{"id":1,"name":"Hari","role":"lead developer","city":"Hyderabad"}'

// JSON string → Object (parse)
const parsed = JSON.parse(jsonString);

jQuery — Write Less, Do More

jQuery is a library that wraps vanilla JS into shorter, cross-browser-safe syntax. First include it in your HTML:

include jQuery
<!-- In <head> or just before </body> -->
<script src="https://code.jquery.com/jquery-3.7.1.min.js"></script>

3.1  jQuery Selectors — The $() function jQuery

Why
$() is jQuery's power tool. It's like document.querySelector but can select anything — IDs, classes, tags, attributes, nth-children — in a single consistent syntax.
selectors.js
// Tag name
$('p')                    // all <p> elements

// ID  (prefix: #)
$('#header')              // element with id="header"

// Class  (prefix: .)
$('.card')                // all elements with class="card"

// Attribute
$('input[type="email"]')  // input whose type is "email"

// Child / descendant
$('#sidebar li')          // all <li> inside #sidebar
$('ul > li')              // direct children only

// First / last / nth
$('.card:first')
$('.card:last')
$('.card:eq(1)')           // second card (zero-indexed)

// Multiple selectors
$('h1, h2, h3')           // all headings at once

// this — the element that triggered the event
$('.card').on('click', function() {
  $(this).addClass('selected');  // only the clicked card
});

AJAX — Communicating with Servers

AJAX (Asynchronous JavaScript And XML) lets your page send and receive data from a server without reloading. This powers real-time search, login checks, saving data, loading content on scroll, etc.

How it works
Your page → sends HTTP request → Server processes → returns JSON → Your JS updates the page. All without a page reload.

4.1  fetch() — Modern native AJAX JS

fetch-get.js
// ── GET request — load data ──────────────────────
fetch('https://jsonplaceholder.typicode.com/users/1')
  .then(response => {
    if (!response.ok) throw new Error("HTTP error: " + response.status);
    return response.json();         // parse JSON
  })
  .then(data => {
    console.log("Name:", data.name);  // "Leanne Graham"
    document.getElementById('result').textContent = data.name;
  })
  .catch(error => {
    console.error("Failed:", error);
  });

// ── Same using async/await (cleaner) ─────────────
async function loadUser(id) {
  try {
    const res  = await fetch(`https://jsonplaceholder.typicode.com/users/${id}`);
    const data = await res.json();
    console.log(data.name, data.email);
    return data;
  } catch (err) {
    console.error("Error:", err);
  }
}
loadUser(1);

4.2  POST Request — Sending data to a server JS

fetch-post.js
async function createPost() {
  const payload = {
    title:  "My New Post",
    body:   "This is the content",
    userId: 1
  };

  const res = await fetch('https://jsonplaceholder.typicode.com/posts', {
    method:  'POST',
    headers: { 'Content-Type': 'application/json' },
    body:    JSON.stringify(payload)    // object → JSON string
  });

  const data = await res.json();
  console.log("Created with ID:", data.id);  // 101
}

$.ajax({

  type:     'GET',                  // or 'POST', 'PUT', 'DELETE'

  beforeSend: function() {
    $('#loader').show();            // show spinner
  },

  success: function(data) {
    $('#loader').hide();
    // Render each user as a list item

      html += `<li><strong>${u.name}</strong> — ${u.email}</li>`;

    $('#userList').html('<ul>' + html + '</ul>');
  },

  error: function(xhr, status, err) {
    $('#loader').hide();
    console.error("AJAX Error:", status, err);
    $('#result').text("Failed to load data.");
  },

  complete: function() {
    console.log("Request finished (success or error)");
  }
});

4.4  Error Handling & Loading States AJAX

full-pattern.js
// Full pattern: loading state + error handling + rendering
async function fetchAndRender(url, containerId) {
  const container = document.getElementById(containerId);

  // Step 1: Show loading state
  container.innerHTML = '<p>⏳ Loading…</p>';

  try {
    // Step 2: Make the request
    const res = await fetch(url);

    // Step 3: Check HTTP status
    if (!res.ok) throw new Error(`Error ${res.status}: ${res.statusText}`);

    // Step 4: Parse
    const data = await res.json();

    // Step 5: Render success
    container.innerHTML = `
      <div class="card">
        <h3>${data.name}</h3>
        <p>${data.email}</p>
      </div>
    `;

  } catch (err) {
    // Step 6: Render error state
    container.innerHTML = `<p style="color:red">❌ ${err.message}</p>`;
  }
}

fetchAndRender('https://jsonplaceholder.typicode.com/users/1', 'result');
💡 HTTP Status Codes you'll encounter
200 OK — success  |  201 Created — POST success  |  400 Bad Request — wrong data sent  |  401 Unauthorized — missing/bad token  |  404 Not Found  |  500 Server Error
