Linking JavaScript to HTML

Three ways to add JavaScript to an HTML page

🟣 Why

HTML gives structure, JS gives behaviour. They must be connected for JS to interact with HTML elements.

🔵 How

Inline, internal <script> tag, or external .js file. External file is best practice for real projects.

🟢 Where

Every single webpage that uses JavaScript — social media, dashboards, e-commerce, forms.

method1_inline.html
<!-- Method 1: INLINE — directly on element -->
<button onclick="alert('Hello!')">
  Click Me
</button>

<!-- ⚠️ OK for tiny tests, avoid in real apps -->


<!-- Method 2: INTERNAL <script> tag -->
<body>
  <h1 id="title">Hello</h1>

  <!-- Script goes at BOTTOM of body -->
  <script>
    document.getElementById('title')
      .textContent = 'Hello World!';
  </script>
</body>


<!-- Method 3: EXTERNAL file (✅ Best Practice) -->
<body>
  <h1 id="title">Hello</h1>

  <!-- defer = run after HTML is loaded -->
  <script src="app.js" defer></script>
</body>
Why script goes at the bottom

HTML loads top to bottom. If <script> is in <head>, it runs before the elements exist — so getElementById returns null.

❌ <head> — element doesn't exist yet

✅ Bottom of <body> — all elements are loaded

✅ <script defer> — loads in parallel, runs after HTML

The Document Object Model — HTML as a JavaScript object tree

// JS navigates the DOM like a tree

querySelector

Select elements using CSS selectors — the most flexible method

querySelector.html
<div class="card">
  <h2 class="title">Card Title</h2>
  <p>Some text</p>
  <button class="btn-primary">Click</button>
</div>
<input type="email" name="email">

<script>
// querySelector — returns FIRST match only

// By tag
document.querySelector("h2");

// By class (add dot)
document.querySelector(".card");
document.querySelector(".btn-primary");

// By id (add #)
document.querySelector("#title");

// By attribute
document.querySelector("input[type='email']");
document.querySelector("[name='email']");

// Nested — .card's <p> child
document.querySelector(".card p");

// First <li> inside a <ul>
document.querySelector("ul > li:first-child");
</script>

querySelectorAll

Select multiple elements at once — returns a NodeList

querySelectorAll.html
<ul id="myList">
  <li class="item">Apple</li>
  <li class="item">Banana</li>
  <li class="item">Mango</li>

<script>
// Get ALL matching elements
const items = document
  .querySelectorAll(".item");

console.log(items.length);  // 3

// Loop through with forEach
items.forEach((li, i) => {
  console.log(`${i}: ${li.textContent}`);

// Change ALL items at once
items.forEach(li => {
  li.style.color = "blue";
  li.style.fontWeight = "bold";
});

// Convert to real array for more methods
const arr = [...items];
const texts = arr.map(li => li.textContent);
console.log(texts);
// ["Apple", "Banana", "Mango"]
</script>

textContent vs innerHTML

Two ways to read and write element content

content.html
<div id="box">
  Hello <strong>World</strong>
</div>

<script>
const box = document.getElementById("box");

// READ
console.log(box.textContent);
// "Hello World"  ← strips HTML tags

console.log(box.innerHTML);
// "Hello <strong>World</strong>"

// ── WRITE ──
// textContent — safe, treats as plain text
box.textContent = "<b>Not bold</b>";
// shows literally: <b>Not bold</b>

// innerHTML — renders HTML tags
box.innerHTML = "<b>This IS bold!</b>";
// shows bold text on page

// Build a list dynamically
const fruits = ["Apple", "Mango", "Banana"];
box.innerHTML = `
  <ul>
    ${fruits.map(f =>
      `<li>${f}</li>`).join("")
    }
  </ul>
`;
</script>
Result on Page

After innerHTML with fruits array:

  • Apple
  • Mango
  • Banana
⚠️
Security: Never put user-typed data into innerHTML — it can run malicious scripts (XSS). Use textContent for user input, innerHTML only for trusted HTML you control.

Changing CSS Styles with JS

Dynamically apply or remove styles on any element

styles.html
<div id="box" style="width:100px;height:100px">
</div>

<script>
const box = document.getElementById("box");

// Set individual style properties
// CSS: background-color → JS: backgroundColor
box.style.backgroundColor = "#f7df1e";
box.style.borderRadius    = "12px";
box.style.padding         = "20px";
box.style.fontSize        = "18px";
box.style.fontWeight      = "bold";
box.style.textAlign       = "center";
box.textContent           = "Styled!";

// Read computed style
const styles = window
  .getComputedStyle(box);
console.log(styles.width);   // "100px"

// Remove a style (set to empty)
box.style.borderRadius = "";
</script>
Result on Page
Styled!
💡
CSS → JS naming: Hyphenated CSS properties become camelCase in JS.
background-color → backgroundColor
font-size → fontSize
border-radius → borderRadius

Working with Attributes

Get, set, and remove HTML attributes like src, href, disabled, placeholder

attributes.html
<img  id="pic"  src="cat.jpg" alt="Cat">
<a    id="link" href="#">Click me</a>
<input id="inp"  type="text">
<button id="btn">Submit</button>

<script>
const img = document.getElementById("pic");
const a   = document.getElementById("link");
const inp = document.getElementById("inp");
const btn = document.getElementById("btn");

// getAttribute — read any attribute
console.log(img.getAttribute("src")); // "cat.jpg"
console.log(img.getAttribute("alt")); // "Cat"

// setAttribute — set any attribute
img.setAttribute("src", "dog.jpg");
img.setAttribute("alt", "Dog");
a.setAttribute("href", "https://google.com");
a.setAttribute("target", "_blank");

// hasAttribute / removeAttribute
console.log(btn.hasAttribute("disabled")); // false
btn.setAttribute("disabled", "");  // disable
btn.removeAttribute("disabled");   // re-enable

// Shorthand for common properties
inp.placeholder = "Enter your name";
inp.disabled    = true;
inp.value       = "Hari";
</script>

Removing Elements

Delete elements from the page dynamically

remove.html
<div id="parent">
  <p id="child">I will be removed</p>
  <p id="keep">I stay</p>
</div>

<script>
// Method 1: .remove() — modern, direct
const child = document
  .getElementById("child");
child.remove();

// Method 2: removeChild — older way
const parent = document
  .getElementById("parent");
const toRemove = document
  .getElementById("keep");
parent.removeChild(toRemove);

// Remove all children at once
parent.innerHTML = "";

// Real example: delete a list item
// when user clicks × button
function addItem(text) {
  const li = document.createElement("li");
  li.textContent = text;
  const btn = document.createElement("button");
  btn.textContent = " ×";
  btn.onclick = () => li.remove();
  li.appendChild(btn);
  list.appendChild(li);
}
</script>

    Keyboard Events

    Detect what keys users press — essential for search, shortcuts, games

    keyboard.html
    <input id="searchBox" placeholder="Type here">
<p id="info"></p>

<script>
const box  = document.getElementById("searchBox");
const info = document.getElementById("info");

// keydown — fires when key pressed
box.addEventListener("keydown", (e) => {
  console.log("Key:", e.key);      // "Enter", "a", " "
  console.log("Code:", e.code);    // "Enter", "KeyA"
  console.log("Ctrl:", e.ctrlKey); // true/false

  if (e.key === "Enter") {
    info.textContent = `Searching: ${box.value}`;
  }
  if (e.key === "Escape") {
    box.value = "";
    info.textContent = "Cleared!";
  }
});

// input — fires on every character change
box.addEventListener("input", (e) => {
  info.textContent = `Typing: ${e.target.value}`;
});

// Global shortcut: Ctrl+S
document.addEventListener("keydown", (e) => {
  if (e.ctrlKey && e.key === "s") {
    e.preventDefault(); // stop browser Save
    console.log("Custom save triggered!");
  }
});
</script>

    Form Events

    Handle form submission, focus, blur, and change events

    formevents.html
    <form id="myForm">
  <input id="name" placeholder="Name">
  <select id="role">
    <option value="user">User</option>
    <option value="admin">Admin</option>
  </select>
  <button type="submit">Submit</button>
</form>

<script>
const form = document.getElementById("myForm");
const name = document.getElementById("name");
const role = document.getElementById("role");

// submit — intercept form submission
form.addEventListener("submit", (e) => {
  e.preventDefault(); // stop page reload
  console.log("Name:", name.value);
  console.log("Role:", role.value);
});

// focus — when user clicks into input
name.addEventListener("focus", () => {
  name.style.borderColor = "blue";
});

// blur — when user leaves the input
name.addEventListener("blur", () => {
  name.style.borderColor = "";
  if (!name.value) name.style.borderColor="red";
});

// change — dropdown changed
role.addEventListener("change", () => {
  console.log("Role changed to:", role.value);
});
</script>

    Timers — setTimeout & setInterval

    Delay code or repeat it on a schedule

    timers.html
    <div id="timer"></div>

<script>
// setTimeout — run ONCE after delay
setTimeout(() => {
  console.log("Runs after 2 seconds");
}, 2000); // ms

// setInterval — run REPEATEDLY
let count = 0;
const id = setInterval(() => {
  count++;
  document.getElementById("timer")
    .textContent = `Count: ${count}`;
}, 1000); // every 1 second

// Stop the interval after 5 seconds
setTimeout(() => {
  clearInterval(id);
  console.log("Stopped!");
}, 5000);

// Real use: countdown timer
function countdown(seconds) {
  let s = seconds;
  const t = setInterval(() => {
    document.getElementById("timer")
      .textContent = `${s}s remaining`;
    if (--s < 0) {
      clearInterval(t);
      document.getElementById("timer")
        .textContent = "🎉 Done!";
    }
  }, 1000);
}
countdown(10);
</script>

    Dark Mode Toggle

    Classic real-world JS + HTML + classList example

    darkmode.html
    <style>
  body {
    background: #fff;
    color: #000;
    transition: all 0.3s;
  }
  body.dark {
    background: #0f1117;
    color: #e2e8f0;
  }
  .toggle-btn {
    padding: 10px 20px;
    cursor: pointer;
    border-radius: 8px;
  }

<button class="toggle-btn" id="darkBtn">
  🌙 Dark Mode
</button>

<script>
const btn = document
  .getElementById("darkBtn");

btn.addEventListener("click", () => {
  document.body.classList.toggle("dark");
  
  const isDark = document.body
    .classList.contains("dark");

  btn.textContent = isDark
    ? "☀️ Light Mode"
    : "🌙 Dark Mode";

  // Remember preference
  localStorage.setItem("theme",
    isDark ? "dark" : "light");
});

// Apply saved preference on load
if (localStorage.getItem("theme") === "dark") {
  document.body.classList.add("dark");
  btn.textContent = "☀️ Light Mode";
}
</script>

    Live Search Filter

    Filter a list in real-time as the user types — no server needed

    livesearch.html
    <input id="search" placeholder="Search...">
<ul id="results"></ul>
<p id="count"></p>

<script>
const cities = [
  "Hyderabad", "Mumbai", "Delhi",
  "Bangalore", "Chennai", "Kolkata",
  "Pune", "Jaipur", "Ahmedabad"
];

const search  = document.getElementById("search");
const results = document.getElementById("results");
const count   = document.getElementById("count");

function render(items) {
  results.innerHTML = items
    .map(c => `<li>${c}</li>`)
    .join("");
  count.textContent = `${items.length} found`;
}

render(cities); // show all on load

search.addEventListener("input", () => {
  const q = search.value.toLowerCase();
  const filtered = cities.filter(c =>
    c.toLowerCase().includes(q)
  );
  render(filtered);
});
</script>

      Fetch API — Load & Display Real Data

      Get data from an API and inject it straight into your HTML

      fetch-html.html
      <div id="userCard">Loading...</div>
<button id="nextBtn">Next User</button>

<script>
let userId = 1;

async function loadUser(id) {
  const card = document
    .getElementById("userCard");

  card.innerHTML = "⏳ Loading...";

  try {
    const res = await fetch(
      `https://jsonplaceholder.typicode.com
        /users/${id}`
    );
    const user = await res.json();

    card.innerHTML = `
      <h3>${user.name}</h3>
      <p>📧 ${user.email}</p>
      <p>🌐 ${user.website}</p>
      <p>🏢 ${user.company.name}</p>
    `;
  } catch (e) {
    card.innerHTML = "❌ Failed to load";
  }
}

loadUser(userId);

document.getElementById("nextBtn")
  .addEventListener("click", () => {
    userId = (userId % 10) + 1;
    loadUser(userId);
  });
</script>
